**Program to calculate the factorial of n**

declare module fibo;

<<<driver program>>>
start
	declare k:integer;
	declare s:integer;
	while(true)
	start
        get_value(k);
        [s]:= use module fibo with parameters k;
        print(s);
    end

end

<<module fibo>>
takes input[n : integer];
returns  [res: integer];
start
    declare t:integer;
	switch(n)
	start
	case 1:
	    res:=1;
	    break;
	case 2:
	    res:=2;
	    break;
	default:
	    n:=n-1;
	    [res]:=use module fibo with parameters n;
	    n:=n-1;
	    [t]:=use module fibo with parameters n;
	    res := res + t;
	    break;
	end
end

<<module factorialN>>
takes input[n : integer];
returns  [res: integer];
start
	switch(n)
	start
	case 0:
	    res:= 1;
	    break;
    default:
        n:=n-1;
        [res]:=use module factorialN with parameters n;
        res:= (n+1)*res;
        break;
	end
end
